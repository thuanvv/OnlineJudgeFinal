# 安装维护

# 🍷 准备工作

  1. 安装docker；参考文档 (opens new window)
sudo curl -fsSL https://get.docker.com | bash -s docker --mirror Aliyun
# 启动docker
sudo systemctl enable docker
sudo systemctl start docker
  1. 安装docker-compose；参考文档 (opens new window)
sudo curl -L "https://github.com/docker/compose/releases/download/v2.2.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose
sudo ln -s /usr/local/bin/docker-compose /usr/bin/docker-compose
docker-compose version

# 🔨 部署

下面以1.3版本为例进行部署（已发行版本列表releases (opens new window)）。

获取配置文件并解压：

wget https://github.com/winterant/OnlineJudge/releases/download/1.3/lduoj-v1.3.zip
unzip lduoj-v1.3.zip

启动服务：

cd lduoj-v1.3
docker-compose up -d
  • 访问首页http://ip:8080；可在宿主机配置域名；
  • 默认管理员用户：admin，默认密码adminadmin，务必更改默认密码；

# 🚗 升级

  • 版本内更新(docker tag不变，无须修改任何配置)，一般会更新一些影响较小的bug修复。
    docker-compose pull web judge
docker-compose up -d
  • 跨版本升级（例如1.3升级到1.4）
    1. 参照【部署】获取新版本并解压；
    wget https://github.com/winterant/OnlineJudge/releases/download/1.4/lduoj-v1.4.zip
unzip lduoj-v1.4.zip
    1. 将旧版本中的./data/文件夹移动到新版本文件夹中；
    mv lduoj-v1.3/data lduoj-v1.4/
    1. 修改必要的配置；
    vim lduoj-v1.4/lduoj.conf
    1. 在新版本文件夹中启动服务即可；
    cd lduoj-v1.4
docker-compose up -d

# 💿 备份/迁移

# 备份

  1. 将docker-compose.yml所在文件夹打包备份；
    tar -cf - ./lduoj | pigz -p $(nproc) > lduoj_bak.tar.gz

# 恢复

  1. 解压备份包
    tar -zxvf lduoj_bak.tar.gz
  2. 一键部署
    cd lduoj_bak
docker-compose up -d
